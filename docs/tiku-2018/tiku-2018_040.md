[路人のhehe](https://www.nowcoder.com/profile/2562288)

```cpp
int bin_search(const int* datas, int len, int key)
{
    int starts = 0;
    int ends = len - 1;
    int middle = int((ends + starts) / 2);
    while(starts <= ends)
    {
        if(datas[middle] == key)
        {
            return middle;
        }
        else if(datas[middle] > key)
        {
            ends = middle - 1;
        }
        else
        {
            starts = middle + 1;
        }
        middle = int((ends + starts) / 2);
    }
    return -1;
}
```
