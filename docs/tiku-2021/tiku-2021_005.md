def solu():
    try:
        n,k = input().split(' ')
    except:
        return False
    n,k = int(n),int(k)
    nums = input().split(' ')
    nums = [int(_) for _ in nums]

    r,l,c,res = 0,0,0,0
    while l<n:
        while c<k and l<n:
            c +=nums[l]
            l+=1
        while c==k and l<n:
            res +=1
            c +=nums[l]
            l +=1
        while c>k and r<l:
            c -= nums[r]
            r +=1
        while c==k and r<l:
            res +=1
            c -=nums[r]
            r +=1
    print(res)
    return True

if __name__=="__main__":
