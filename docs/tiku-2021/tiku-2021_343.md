    private void dfs(char[][] grid, int x, int y) {

        dfs(grid, x - 1, y);
        dfs(grid, x, y - 1);
        dfs(grid, x, y + 1);

[零葬](https://www.nowcoder.com/profile/75718849)

就是给最长递增子序列套了个业务背景，我们可以将时间复杂度为 O(n²)的经典动态规划方法优化成 O(nlogn)的算法

```cpp
import java.util.*;

public class Solution {
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     * 
     * @param nums int 整型一维数组 
     * @return int 整型
     */
    public int lengthOfLIS (int[] nums) {
        // write code here
        int n = nums.length;
        int[] ends = new int[n];    // ends[i]表示长度为 i+1 的递增子序列中的最小结尾
        ends[0] = nums[0];
        int tail = 0, maxLen = 1;
        for(int i = 1; i < n; i++){
            int index = lowerBound(ends, 0, tail, nums[i]);
            ends[index] = nums[i];
            if(index > tail){
                tail ++;
            }
            maxLen = Math.max(maxLen, index + 1);
        }
        return maxLen;
    }

    private int lowerBound(int[] nums, int L, int R, int target) {
        int left = L, right = R, index = R + 1;
        while(left <= right){
            int mid = left + ((right - left) >> 1);
            if(nums[mid] < target){
                left = mid + 1;
            }else{
                index = mid;
                right = mid - 1;
            }
        }
        return index;
    }
}
```

        # write code here
        dp=[1 for i in range(len(nums))]

        for i in range(1,len(nums)):
            for j in range(i):
                if nums[i]>nums[j]:
                    dp[i]=max(dp[i],dp[j]+1)
        return max(dp)
