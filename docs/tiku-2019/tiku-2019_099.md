class Solution(object):
    def lights(self, num, _str):
        if num == 0:
            return 0
        pos = 0
        light = 0
        while pos < len(_str):
            if _str[pos] == '.':
                light += 1
                pos += 3
                if pos >= len(_str):
                    break
            if _str[pos] == 'X':
                pos += 1
        return light

def main():
